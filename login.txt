        try{
            Class.forName("com.mysql.jdbc.Driver");
            
            Connection con = DriverManager.getConnection("jdbc:mysql://localhost:3306/bakery","root","mysql");
            
            String sql = "Select * from bakery.bakery_login  where username=? and password=?"; 
            
            PreparedStatement pst = con.prepareStatement(sql);

            pst.setString(1, username.getText());

            pst.setString(2, password.getText());

            ResultSet rs = pst.executeQuery();

            if(rs.next()){

            JOptionPane.showMessageDialog(null, "Username and Password Matched");
         
            bakery_purchase s = new bakery_purchase();
            s.setVisible(true);
            
            } 
            else{

            JOptionPane.showMessageDialog(null, "Username and password not Correct");

            username.setText("");

            password.setText("");

            }

            con.close();

        }
        catch(SQLException e){
            JOptionPane.showMessageDialog(null, e);
        } catch (ClassNotFoundException ex) {
            Logger.getLogger(bakery_login.class.getName()).log(Level.SEVERE, null, ex);
        }
